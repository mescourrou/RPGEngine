72 void CharacterGUI::prepare(const sf::Vector2f& targetSize)
73 {
74  if (m_tics == 0)
75  {
76  auto actualiseCurrentSprite = [this](const std::vector<unsigned int>& action)
77  {
78  m_currentSprite = &(m_sprites[action.at(m_spriteCinematicIndex)]);
79  m_spriteCinematicIndex++;
80  if (m_spriteCinematicIndex >= action.size())
81  m_spriteCinematicIndex = 0;
82  };
83  if (m_moving && m_currentDirection == Left)
84  actualiseCurrentSprite(m_actions[actions::LEFT]);
85  else if (m_moving && m_currentDirection == Right)
86  actualiseCurrentSprite(m_actions[actions::RIGHT]);
87  else if (m_moving && m_currentDirection == Down)
88  actualiseCurrentSprite(m_actions[actions::DOWN]);
89  else if (m_moving && m_currentDirection == Up)
90  actualiseCurrentSprite(m_actions[actions::UP]);
91  // If not moving, we cycle the stopped sprites
92  if (!m_moving)
93  {
94  switch (m_currentDirection)
95  {
96  case Left:
97  actualiseCurrentSprite(m_actions[actions::LEFT_STOPPED]);
98  break;
99  case Right:
100  actualiseCurrentSprite(m_actions[actions::RIGHT_STOPPED]);
101  break;
102  case Up:
103  actualiseCurrentSprite(m_actions[actions::UP_STOPPED]);
104  break;
105  case Down:
106  actualiseCurrentSprite(m_actions[actions::DOWN_STOPPED]);
107  break;
108  }
109  }
110  }
111 
112  setOnScreenPosition(m_map.lock()->positionOnScreenFrom(
113  m_character.lock()->position()));
114  m_tics++;
115  if (m_tics >= m_spriteChangeTics)
116  m_tics = 0;
117 }

202

204  unsigned int width = set[characterFile::WIDTH].get<unsigned int>();
205

220  return false;

222  set[characterFile::BACKGROUND].at(1), set[characterFile::BACKGROUND].at(2));

228

242  }

254  }
255  m_actions[a.key()] = spriteList;
256  }
257 
258  for (const auto& actionName : m_requiredActions)
259  {
260  if (m_actions.find(actionName) == m_actions.end())
261  return false;
262  }
263 
264  // Initialize the first sprite
265  m_currentSprite = &(m_sprites[m_actions[actions::DOWN_STOPPED].front()]);
266 
267  }
268  else
269  LOG(WARNING) << "Warning : Impossible to open " << characterRessourcesDir <<
270  "/" + Tools::snakeCase(m_character.lock()->name()) << ".json";
271
