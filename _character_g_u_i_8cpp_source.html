74 void CharacterGUI::prepare(const sf::Vector2f&)
75 {
76  PROFILE_FUNCTION();
77  if (m_tics == 0)
78  {
79  auto actualiseCurrentSprite = [this](const std::vector<unsigned int>& action)
80  {
81  m_currentSprite = &(m_sprites[action.at(m_spriteCinematicIndex)]);
82  m_spriteCinematicIndex++;
83  if (m_spriteCinematicIndex >= action.size())
84  m_spriteCinematicIndex = 0;
85  };
86  if (m_moving && m_currentDirection == Left)
87  actualiseCurrentSprite(m_actions[actions::LEFT]);
88  else if (m_moving && m_currentDirection == Right)
89  actualiseCurrentSprite(m_actions[actions::RIGHT]);
90  else if (m_moving && m_currentDirection == Down)
91  actualiseCurrentSprite(m_actions[actions::DOWN]);
92  else if (m_moving && m_currentDirection == Up)
93  actualiseCurrentSprite(m_actions[actions::UP]);
94  // If not moving, we cycle the stopped sprites
95  if (!m_moving)
96  {
97  switch (m_currentDirection)
98  {
99  case Left:
100  actualiseCurrentSprite(m_actions[actions::LEFT_STOPPED]);
101  break;
102  case Right:
103  actualiseCurrentSprite(m_actions[actions::RIGHT_STOPPED]);
104  break;
105  case Up:
106  actualiseCurrentSprite(m_actions[actions::UP_STOPPED]);
107  break;
108  case Down:
109  actualiseCurrentSprite(m_actions[actions::DOWN_STOPPED]);
110  break;
111  default:
112  throw BaseException("Unknown direction");
113  break;
114  }
115  }
116  }
117 
118  setOnScreenPosition(currentMap().lock()->positionOnScreenFrom(
119  m_character.lock()->position()));
120  m_tics++;
121  if (m_tics >= m_spriteChangeTics)
122  m_tics = 0;
123 }

210 bool CharacterGUI::loadSets(const json& json,
211  const std::string& characterRessourcesDir)
212 {
213  PROFILE_FUNCTION();
214  namespace characterFile = config::structure::characterFile;
215  for (auto set : json[characterFile::SPRITE_SETS])
216  {
217  if (!verifyJSONSpriteSetsStructure(set))
218  return false;
219 
220  unsigned int id = 0;
221 
222  id = set[characterFile::FIRST_ID].get<unsigned int>();
223 
224  std::string filename = characterRessourcesDir + '/' +
225  set[characterFile::SET_FILE].get<std::string>();
226  if (!std::filesystem::exists(filename))
227  return false;
228  sf::Image image;
229  image.loadFromFile(filename);
230 
231  unsigned int height = set[characterFile::HEIGHT].get<unsigned int>();
232  unsigned int width = set[characterFile::WIDTH].get<unsigned int>();
233 
234  unsigned int spriteHeight =
235  set[characterFile::SPRITE_HEIGHT].get<unsigned int>();
236  unsigned int spriteWidth = set[characterFile::SPRITE_WIDTH].get<unsigned int>();
237 
238  unsigned int hShift = set[characterFile::HORIZONTAL_SHIFT].get<unsigned int>();
239  unsigned int vShift = set[characterFile::VERTICAL_SHIFT].get<unsigned int>();
240 
241  sf::Vector2f origin(set[characterFile::ORIGIN_X].get<unsigned int>(),
242  set[characterFile::ORIGIN_Y].get<unsigned int>());
243  // Remove backgroung if specified
244  if (set.contains(characterFile::BACKGROUND)
245  && set[characterFile::BACKGROUND].is_array())
246  {
247  if (set[characterFile::BACKGROUND].size() != 3)
248  return false;
249  sf::Color backgroundColor(set[characterFile::BACKGROUND].at(0),
250  set[characterFile::BACKGROUND].at(1), set[characterFile::BACKGROUND].at(2));
251  image.createMaskFromColor(backgroundColor);
252  }
253 
254  m_textures.push_back(std::make_shared<sf::Texture>());
255  m_textures.back()->loadFromImage(image);
256 
257  for (unsigned int i = 0; i < height ; i++)
258  {
259  for (unsigned int j = 0; j < width; j++)
260  {
261  m_sprites.emplace(std::pair<unsigned int, sf::Sprite>(id,
262  sf::Sprite(*m_textures.back(),
263  sf::IntRect(hShift + j * spriteWidth, vShift + i * spriteHeight,
264  spriteWidth, spriteHeight))));
265  m_sprites[id].setOrigin(origin);
266  id++;
267  }
268  }
269 
270  }
271  return true;
272 }
