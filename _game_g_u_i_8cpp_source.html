116 void GameGUI::eventManager()
117 {
118  static sf::Clock deltaClock;
119  ImGui::SFML::Update(*m_window, deltaClock.restart());
120  // Process events
121  while (m_window->pollEvent(m_event))
122  {
123  ImGui::SFML::ProcessEvent(m_event);
124  // Close window: exit
125  if (m_event.type == sf::Event::Closed)
126  {
127  m_signalOnClose.trigger();
128  }
129  if (m_event.type == sf::Event::KeyPressed)
130  {
131  if (m_actionWaitingForKeybinding.empty())
132  {
133  signalKeyPressed.trigger(m_event.key);
134  events::ActionHandler::processSFMLEvent(m_event.key);
135  }
136  }
137  if (m_event.type == sf::Event::KeyReleased)
138  {
139  if (m_actionWaitingForKeybinding.empty())
140  {
141  switch (m_event.key.code)
142  {
143  case sf::Keyboard::Escape:
144  signalPause.trigger(!m_ui.onPause);
145  break;
146  case sf::Keyboard::U:
147  m_ui.uiActivated = !m_ui.uiActivated;
148  break;
149  }
150  signalKeyReleased.trigger(m_event.key);
151  }
152  else
153  {
154  auto keyBinding = events::KeyBinding::fromSFML(m_event.key);
155  if (!keyBinding.isKey(events::KeyBinding::NOT_BINDED, events::KeyBinding::NONE))
156  {
157  events::ActionHandler::setKeyBinding(m_actionWaitingForKeybinding, keyBinding);
158  m_actionWaitingForKeybinding = "";
159  }
160  }
161  }
162  }
163 
164  if (sf::Keyboard::isKeyPressed(sf::Keyboard::Left))
165  signalArroyIsPressed.trigger(sf::Keyboard::Left);
166  if (sf::Keyboard::isKeyPressed(sf::Keyboard::Right))
167  signalArroyIsPressed.trigger(sf::Keyboard::Right);
168  if (sf::Keyboard::isKeyPressed(sf::Keyboard::Down))
169  signalArroyIsPressed.trigger(sf::Keyboard::Down);
170  if (sf::Keyboard::isKeyPressed(sf::Keyboard::Up))
171  signalArroyIsPressed.trigger(sf::Keyboard::Up);
172 
173  makeUI();
174 
175 }
