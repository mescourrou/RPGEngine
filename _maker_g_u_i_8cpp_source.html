116 void MakerGUI::eventManager()
117 {
118  static sf::Clock deltaClock;
119  ImGui::SFML::Update(m_window, deltaClock.restart());
120  // Process events
121  sf::Event event;
122  while (m_window.pollEvent(event))
123  {
124  ImGui::SFML::ProcessEvent(event);
125  // Close window: exit
126  if (event.type == sf::Event::Closed)
127  {
128  events::ActionHandler::execute("Quit");
129  }
130  if (event.type == sf::Event::Resized)
131  {
132  if (m_mapGUI) m_mapGUI->forcePrepare(m_window.getView().getSize());
133  }
134  if (event.type == sf::Event::KeyPressed)
135  {
136  events::ActionHandler::processSFMLEvent(event.key);
137  switch (event.key.code)
138  {
139  case sf::Keyboard::Left:
140  if (m_mapGUI) m_mapGUI->move(-5, 0);
141  break;
142  case sf::Keyboard::Right:
143  if (m_mapGUI) m_mapGUI->move(5, 0);
144  break;
145  case sf::Keyboard::Up:
146  if (m_mapGUI) m_mapGUI->move(0, -5);
147  break;
148  case sf::Keyboard::Down:
149  if (m_mapGUI) m_mapGUI->move(0, 5);
150  break;
151  default:
152  break;
153  }
154  }
155  if (event.type == sf::Event::KeyReleased)
156  {
157  switch (event.key.code)
158  {
159  case sf::Keyboard::Escape:
160  break;
161  default:
162  break;
163  }
164  }
165  }
166 
167  makeUI();
168 }
