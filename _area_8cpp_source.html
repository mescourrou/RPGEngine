160 bool Area::intersect(const Vector<2>& origin, const Vector<2>& vector,
161  Vector<2>& outIntersect) const
162 {
163  auto destination = origin + vector;
164  if (vector == Vector<2> {0, 0})
165  return false;
166 
167  // Parametrize vectors as
168  // x = alpha * a + x_0
169  // y = alpha * b + y_0
170  // alpha in [0, 1]
171 
172  double a1 = vector.x();
173  double b1 = vector.y();
174 
175  bool intersection = false;
176

178  {
179  //const auto& pt0 = (i == 0) ? m_points.at(m_points.size()-1) : m_points.at(i-1);
180  const auto& pt1 = m_points.at(i);
181  const auto& pt2 = (i == m_points.size() - 1) ? m_points.at(0) : m_points.at(
182  i + 1);
183  const auto vector2 = pt2 - pt1;
184  if (vector2 == Vector<2> {0, 0})
185  continue;
186  auto vect1Normalized = vector / vector.norm();
187  auto vect2Normalized = vector2 / vector2.norm();
188 
189  bool solved = false;
190  bool isCandidate = false;
191  Vector<2> candidate;
192 
193  if (vect1Normalized == vect2Normalized) // Parallelism
194  {
195  if (onSegment(pt1, origin, pt2))
196  {
197  continue;
198  }
199  else if (onSegment(pt1, destination, pt2))
200  {
201  if ((pt1 - origin).norm() < (pt2 - origin).norm())
202  candidate = pt1;
203  else
204  candidate = pt2;
205  isCandidate = true;
206  solved = true;
207  }
208  }
209 
210  double a2 = vector2.x();
211  double b2 = vector2.y();
212 
213  if (!solved && std::abs(b1) > ZERO_EPSILON)
214  {
215  double denominator = a2 - b2 / b1 * a1;
216  if (std::abs(denominator) > ZERO_EPSILON) // Denominator not null
217  {
218  double alpha2 = ((pt1.y() - origin.y()) / b1 * a1 + origin.x() - pt1.x()) /
219  denominator;
220  if (alpha2 >= 0 && alpha2 <= 1)
221  {
222  double alpha1 = (alpha2 * b2 + pt1.y() - origin.y()) / b1;
223  if (alpha1 >= 0 && alpha1 <= 1) // Vectors intersects
224  {
225  candidate = {alpha1* a1 + origin.x(),
226  alpha1* b1 + origin.y()
227  };
228  isCandidate = true;
229 
230  }
231  }
232  solved = true;
233  }
234  }
235

237  {

239  if (std::abs(denominator) > ZERO_EPSILON) // Denominator not null

354 bool Area::intersectYHalfLine(const Vector<2>& pt1, const Vector<2>& pt2,
355  const Vector<2>& origin, Direction dir)
356 {
357  const auto& highPt = std::max(pt1, pt2, [](const Vector<2>& p1,
358  const Vector<2>& p2)
359  {
360  return p1.y() < p2.y();
361  });
362  const auto& lowPt = std::min(pt1, pt2, [](const Vector<2>& p1,
363  const Vector<2>& p2)
364  {
365  return p1.y() <= p2.y();
366  });
367 
368  if (highPt.y() >= origin.y() && lowPt.y() <= origin.y())
369  {
370  Vector<2> intersect{0, origin.y()};
371  if (std::abs(lowPt.y() - highPt.y()) < ZERO_EPSILON)
372  intersect.x() = pt1.x();
373  else
374  {
375  double relativePosition = 1 - (highPt.y() - origin.y()) /
376  (highPt.y() - lowPt.y());
377  const auto& rightyPt = std::max(pt1, pt2, [](const Vector<2>& p1,
378  const Vector<2>& p2)
379  {
380  return p1.x() < p2.x();
381  });
382  const auto& leftyPt = std::min(pt1, pt2, [](const Vector<2>& p1,
383  const Vector<2>& p2)
384  {
385  return p1.x() <= p2.x();
386  });
387  intersect.x() = leftyPt.x() + relativePosition * (rightyPt.x() - leftyPt.x());
388  }
389
