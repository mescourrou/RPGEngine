156 bool Map::loadCollisionLayer(const json& layer)
157 {
158  if (!layer.is_object())
159  return false;
160  namespace mapFile = config::structure::mapFile;
161  if (!layer[mapFile::KEY_OBJECTS].is_array())
162  return false;
163  for (auto& polygon : layer[mapFile::KEY_OBJECTS])
164  {
165  if (!polygon.is_object())
166  return false;
167  if (!polygon.contains(mapFile::KEY_VISIBLE)
168  || !polygon[mapFile::KEY_VISIBLE].is_boolean())
169  return false;
170  // If the polygon is visible
171  if (polygon[mapFile::KEY_VISIBLE].get<bool>())
172  {
173  if (!polygon.contains(mapFile::KEY_X) || !polygon.contains(mapFile::KEY_Y) ||
174  !polygon[mapFile::KEY_X].is_number() || !polygon[mapFile::KEY_Y].is_number())
175  return false;
176  Vector<2> polygonPosition;
177  polygonPosition.x() = polygon[mapFile::KEY_X].get<double>();
178  polygonPosition.y() = polygon[mapFile::KEY_Y].get<double>();
179 
180  if (!polygon.contains(mapFile::KEY_HEIGHT)
181  || !polygon.contains(mapFile::KEY_WIDTH) ||
182  !polygon[mapFile::KEY_HEIGHT].is_number()
183  || !polygon[mapFile::KEY_WIDTH].is_number())
184  return false;
185 
186  // Polygon
187  if (polygon[mapFile::KEY_HEIGHT].get<unsigned int>() == 0
188  && polygon[mapFile::KEY_WIDTH].get<unsigned int>() == 0)
189  {
190  if (!polygon[mapFile::KEY_POLYGON].is_array())
191  {
192  return false;
193  }
194  Area a;
195  for (auto& point : polygon[mapFile::KEY_POLYGON])
196  {
197  if (!polygon.contains(mapFile::KEY_X) || !polygon.contains(mapFile::KEY_Y) ||
198  !polygon[mapFile::KEY_X].is_number() || !polygon[mapFile::KEY_Y].is_number())
199  return false;
200  a.addPoint(polygonPosition + Vector<2> {point[mapFile::KEY_X].get<double>(), point[mapFile::KEY_Y].get<double>()});
201  }
202  addCollisionArea(a);
203  }
204  else // Rectangle
205  {
206  unsigned int height = polygon[mapFile::KEY_HEIGHT].get<unsigned int>();
207  unsigned int width = polygon[mapFile::KEY_WIDTH].get<unsigned int>();
208  Area a{polygonPosition,
209  {polygonPosition.x(), polygonPosition.y() + height},
210  {polygonPosition.x() + width, polygonPosition.y() + height},
211  {polygonPosition.x() + width, polygonPosition.y()}
212  };
213  addCollisionArea(a);
214 
215  }
216  }
217  }
218 
219  return true;
220 }
