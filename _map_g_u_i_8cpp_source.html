101 void MapGUI::prepare(const sf::Vector2f& targetSize)
102 {
103  PROFILE_FUNCTION();
104  if (m_mapMoved) // We compute the parameters only if the map moved since last time
105  {
106  // Position on the map of the top left corner of the screen
107  // Unit : pixels
108  m_topLeftPosition.x = m_centerOfView.x() - targetSize.x / 2.0;
109  m_topLeftPosition.y = m_centerOfView.y() - targetSize.y / 2.0;
110  // Coordinates of the tile of the top left corner

112  m_chunksWidthPixels);
113  m_firstTileCoordinates.y = std::floor(m_topLeftPosition.y /
114  m_chunksHeightPixels);
115 
116  // Position on the screen of the top left displayed tile
117  // It's negative to cover all the screen
118  int sign = -1;
119  if (m_topLeftPosition.x > 0)
120  sign = 1;
121  m_origin.x = - Tools::linearModulo(m_topLeftPosition.x,
122  static_cast<float>(m_chunksWidthPixels));
123  sign = -1;
124  if (m_topLeftPosition.y > 0)
125  sign = 1;
126  m_origin.y = - Tools::linearModulo(m_topLeftPosition.y,
127  static_cast<float>(m_chunksHeightPixels));
128 
129  m_mapMoved = false;
130  }

197 bool MapGUI::loadTiles(const json& layer)
198 {
199  PROFILE_FUNCTION();
200  namespace mapFile = config::structure::mapFile;
201  if (!layer.contains(mapFile::KEY_TILE_DATA))
202  return false;
203  if (!layer[mapFile::KEY_TILE_DATA].is_array())
204  return false;
205 
206  unsigned int i = 0;
207  unsigned int j = 0;
208  sf::Vector2f currentDrawingPosition;
209  std::map<unsigned int, std::map<unsigned int, unsigned int>> map;
210  for (auto& id : layer[mapFile::KEY_TILE_DATA])
211  {
212  map[j][i] = id;
213 
214  j++;
215  if (j >= m_height)
216  {
217  j = 0;
218  i++;
219  }
220  }
221  for (unsigned int bigI = 0; bigI < std::ceil(m_nbChunksWidth); bigI++)
222  {
223  for (unsigned int bigJ = 0; bigJ < std::ceil(m_nbChunksHeight); bigJ++)
224  {
225  auto& chunkTexture = m_chunksTextures.emplace_back(
226  std::make_shared<sf::RenderTexture>());
227  unsigned int currentWidth = std::min(CHUNK_WIDTH,
228  (m_width - bigI * CHUNK_WIDTH));
229  unsigned int currentHeight = std::min(CHUNK_HEIGHT,
230  (m_height - bigJ * CHUNK_HEIGHT));
231  chunkTexture->create(currentWidth * m_tileWidth, currentHeight * m_tileHeight);
232 
233  for (i = 0; i < currentWidth; i++)
234  {
235  for (j = 0; j < currentHeight; j++)
236  {
237  sf::Sprite sprite(m_tiles.at(map[bigI * CHUNK_WIDTH + i][bigJ * CHUNK_HEIGHT +
238  j]));
239  sprite.setPosition(i * m_tileWidth, j * m_tileHeight);
240  chunkTexture->draw(sprite);
241  }
242  }
243  sf::Sprite sprite(chunkTexture->getTexture());
244  sprite.setOrigin(0, sprite.getGlobalBounds().height);
245  sprite.setScale(1, -1);
246  m_chunks[bigJ][bigI] = std::move(sprite);
247  }
248  }
249 
250  return true;
251 }

261 bool MapGUI::loadTilesets(const std::string& mapDirPath, const json& json)

278  return false;

283  return false;

286

324 bool MapGUI::loadTileset(const std::string& mapDirPath, const json& tileset)
325 {
326  PROFILE_FUNCTION();
327  VLOG(verbosityLevel::FUNCTION_CALL) << "loadTileset";
328  namespace mapFile = config::structure::mapFile;
329  if (!tileset.is_object())
330  return false;
331  if (!tileset.contains(mapFile::KEY_TILESET_FIRST_ID)
332  || !tileset[mapFile::KEY_TILESET_FIRST_ID].is_number_integer())
333  return false;
334  unsigned int id = tileset[mapFile::KEY_TILESET_FIRST_ID].get<unsigned int>();
335 
336  if (!tileset.contains(mapFile::KEY_TILESET_SOURCE)
337  || !tileset[mapFile::KEY_TILESET_SOURCE].is_string())
338  return false;
339 
340  namespace tilesetFile = config::structure::tilesetFile;
341  tinyxml2::XMLDocument doc;
342  tinyxml2::XMLError err = doc.LoadFile(std::string(mapDirPath + '/' +
343  tileset[mapFile::KEY_TILESET_SOURCE].get<std::string>()).c_str());
344  if (err)
345  {
346  LOG(ERROR) << "Error during loading file " <<
347  tileset[mapFile::KEY_TILESET_SOURCE].get<std::string>().c_str();
348  }
349 
350  auto xmlTileset = doc.FirstChildElement(tilesetFile::ELEMENT_TILESET);
351  if (!xmlTileset)
352  {
353  LOG(ERROR) << "No " << tilesetFile::ELEMENT_TILESET << " in the Tileset " <<
354  mapDirPath << '/' << tileset[mapFile::KEY_TILESET_SOURCE].get<std::string>();
355  return false;
356  }
357  if (!xmlTileset->FindAttribute(tilesetFile::PROPERTY_TILE_WIDTH))
358  {
359  LOG(ERROR) << "No " << tilesetFile::PROPERTY_TILE_WIDTH << " in the Tileset " <<
360  mapDirPath << '/' << tileset[mapFile::KEY_TILESET_SOURCE].get<std::string>();
361  return false;
362  }
363  unsigned int width = xmlTileset->FindAttribute(
364  tilesetFile::PROPERTY_TILE_WIDTH)->IntValue();
365 
366  if (!xmlTileset->FindAttribute(tilesetFile::PROPERTY_TILE_HEIGHT))
367  {
368  LOG(ERROR) << "No " << tilesetFile::PROPERTY_TILE_HEIGHT << " in the Tileset "
369  << mapDirPath << '/' << tileset[mapFile::KEY_TILESET_SOURCE].get<std::string>();
370  return false;
371  }
372  unsigned int height = xmlTileset->FindAttribute(
373  tilesetFile::PROPERTY_TILE_HEIGHT)->IntValue();
374 
375  if (!xmlTileset->FindAttribute(tilesetFile::PROPERTY_TILE_COUNT))
376  {
377  LOG(ERROR) << "No " << tilesetFile::PROPERTY_TILE_COUNT << " in the Tileset " <<
378  mapDirPath << '/' << tileset[mapFile::KEY_TILESET_SOURCE].get<std::string>();
379  return false;
380  }
381  unsigned int tilecount = xmlTileset->FindAttribute(
382  tilesetFile::PROPERTY_TILE_COUNT)->IntValue();
383 
384  if (!xmlTileset->FindAttribute(tilesetFile::PROPERTY_TILE_COLUMNS))
385  {
386  LOG(ERROR) << "No " << tilesetFile::PROPERTY_TILE_COLUMNS << " in the Tileset "
387  << mapDirPath << '/' << tileset[mapFile::KEY_TILESET_SOURCE].get<std::string>();
388  return false;
389  }
390  unsigned int columns = xmlTileset->FindAttribute(
391  tilesetFile::PROPERTY_TILE_COLUMNS)->IntValue();
392 
393  auto xmlImage = xmlTileset->FirstChildElement(tilesetFile::ELEMENT_IMAGE);
394  std::string imageFilename = xmlImage->FindAttribute(
395  tilesetFile::PROPERTY_IMAGE_SOURCE)->Value();
396  imageFilename = mapDirPath + '/' + imageFilename;
397 
398  m_textures.emplace_back(std::make_shared<sf::Texture>());
399  if (!m_textures.back()->loadFromFile(imageFilename))
400  LOG(WARNING) << "Texture from " << imageFilename << " not loaded";
401 
402  for (unsigned int line = 0; ; line++)
403  {
404  for (unsigned int col = 0; col < columns; col++)
405  {
406  sf::Sprite tile(*m_textures.back(), sf::IntRect(col * width, line * height,
407  width, height));
408  m_tiles.insert(std::pair<unsigned int, sf::Sprite>((const unsigned int)id,
409  tile));
410 
411  id++;
412  if (id > tilecount)
413  break;
414  }
415  if (id > tilecount)
416  break;
417  }
418 
419 
420  LOG(INFO) << "Loading map succefully";
421  return true;
422 }
